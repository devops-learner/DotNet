 private bool disposed = false;
  
  public void Dispose()
  {
    Dispose(true);
    GC.SuppressFinalize(this);
  }
  
  ~Repository_General()
  {
    Dispose(false);
  }
  
  protected virtual void Dispose(bool disposing)
  {
    if (!this.disposed)
    {
      if (disposing)
      {
        if (this.db != null)
          {
            this.db.Dispose();
            this.db = null;
          }
      }
    }
  }
